        def timestamp(time) # Pretty timestamps!
                hour = (time.hour<10 ? "0"+time.hour.to_s : time.hour)
                minute = (time.min<10 ? "0"+time.min.to_s : time.min)
                second = (time.sec<10 ? "0"+time.min.to_s : time.sec)
                thetime = "[#{hour}:#{minute}:#{second}]"
                return thetime
        end
