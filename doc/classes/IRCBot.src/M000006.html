        def main_loop()
                while @connected
                        ready = select([@irc],nil,nil,0.5)
                        if ready then
                                return if @irc.eof
                                s = @irc.gets
                                rawlog(s)
                                handle_server_input(s)
                        end
                        handle_queue()       
                end
        end
